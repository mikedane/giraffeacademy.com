{{< code lang="sql" >}}

FROM student;

SELECT student.name, student.major
FROM student;

SELECT *
FROM student
WHERE name = 'Jack';

SELECT *
FROM student
WHERE student_id > 2;

FROM student
WHERE major = 'Biology' AND student_id > 1;
