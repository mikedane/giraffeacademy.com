          string rating;

               this->setRating(rating);
          }

          string getRating(){
               return this->rating;
          }
          void setRating(string rating){
               if(rating == "G" || rating == "PG" || rating == "PG-13" || rating == "R" || rating == "NR" || ){
                    this->rating = rating;
               } else {
                    this->rating = "NR";
               }     
          }
